def answer_six():

    import operator    

    PR = nx.pagerank(G2,alpha=0.85)

    sorted_x = sorted(PR.items(), key=operator.itemgetter(1),reverse=True)[0:5]

    result_arr = []                

    for x in range(0,5):

        print ((sorted_x)[x][0])

        result_arr.append((sorted_x)[x][0])
